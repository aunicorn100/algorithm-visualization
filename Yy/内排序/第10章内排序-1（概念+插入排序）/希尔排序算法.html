// 希尔排序算法
void ShellSort(RecType R[], int n)
{   int i, j, d;
    RecType tmp;
    d = n / 2;              // 增量置初值
    while (d > 0) {         // 当增量大于0时循环
        for (i = d; i < n; i++) { // 对相隔d位置的元素组直接插入排序
            tmp = R[i];      // 保存当前元素
            j = i - d;       // 前一个同组元素位置
            while (j >= 0 && tmp.key < R[j].key) {
                R[j + d] = R[j]; // 元素后移
                j = j - d;   // 向前比较
            }
            R[j + d] = tmp;  // 插入到正确位置
        }
        d = d / 2;           // 减小增量
    }
}
// 平均时间复杂度：优于O(n²)
