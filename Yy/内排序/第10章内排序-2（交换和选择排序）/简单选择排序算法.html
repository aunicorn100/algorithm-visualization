// 简单选择排序算法
void SelectSort(RecType R[], int n)
{
    int i, j, k;
    for (i=0; i<n-1; i++)      // 做第i趟排序
    {
        k = i;
        for (j=i+1; j<n; j++)
            if (R[j].key < R[k].key)
                k = j;
        if (k != i)             // R[i] <=> R[k]
            swap(R[i], R[k]);
    }
}
