// 冒泡排序算法
void BubbleSort(RecType R[], int n)
{   int i, j;   RecType temp;
    for (i=0; i<n-1; i++)
    {
        for (j=n-1; j>i; j--)
            // 比较找本趟最小关键字的元素
            if (R[j].key < R[j-1].key)  // 反序
                swap(R[j], R[j-1]);     // R[j] <=> R[j-1]
    }
}
// 时间复杂度：O(n²)
