// 以a[low]为基准划分
int Partition(int a[], int low, int high)
{
    int i = low, j = high;
    int pivot = a[low];
    while (i < j)
    {
        while (i < j && a[j] >= pivot)
            j--;
        a[i] = a[j];
        while (i < j && a[i] <= pivot)
            i++;
        a[j] = a[i];
    }
    a[i] = pivot;
    return i;
}

// 求解算法
int Solution(int a[], int n)
{
    int low = 0, high = n - 1;
    bool flag = true;
    while (flag)
    {
        int i = Partition(a, low, high);
        if (i == n/2 - 1)      // 基准a[i]为第n/2的元素
            flag = false;
        else if (i < n/2 - 1)  // 在右区间查找
            low = i + 1;
        else                   // 在左区间查找
            high = i - 1;
    }

    int s1 = 0, s2 = 0;
    for (int i = 0; i < n/2; i++)
        s1 += a[i];
    for (int j = n/2; j < n; j++)
        s2 += a[j];
    return s2 - s1;
}
