算法核心思想

  • 使用KMP算法高效查找模式串在主串中的所有出现位置
  • 通过预处理模式串生成next数组，避免主串指针回溯
  • 当字符不匹配时，根据next数组移动模式串指针，提高匹配效率
  • 时间复杂度为O(m+n)，其中m为主串长度，n为模式串长度
  • StrCount2函数用于统计模式串在主串中出现的次数
  • GetNext函数用于计算模式串的next数组

int StrCount2(SqString s, SqString t) //利用KMP算法求t在s中出现的次数
{
    int i = 0, j = 0, count = 0;       //定义变量
    int next[MaxSize];                 //定义next数组
    GetNext(t, next);                  //计算next数组
    while (i < s.length && j < t.length) //循环匹配
        if (j == -1 || s.data[i] == t.data[j]) //字符匹配
        {
            i++;                         //主串和子串依次匹配下一个字符
            j++;
        }
        else                             //字符不匹配
            j = next[j];                 //子串根据next数组回退
    if (j >= t.length)                   //匹配成功
    {
        count++;                         //计数器加1
        j = 0;                           //重置j为0，继续匹配
    }
    return count;                        //返回匹配次数
}

void GetNext(SqString t, int next[])
{
    int j, k;
    j = 0; k = -1; next[0] = -1;       //初始化
    while (j < t.length - 1)
        if (k == -1 || t.data[j] == t.data[k])
        {
            j++; k++;
            next[j] = k;
        }
        else
            k = next[k];
}
