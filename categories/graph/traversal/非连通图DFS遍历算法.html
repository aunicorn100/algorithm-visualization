核心思想（匹配题目）

  • 全局visited[]数组置初值0，标记顶点访问状态
  • 从起始顶点开始，递归执行DFS遍历
  • 遍历结束后，检查所有visited[i]是否为1
  • 全为1则图连通（返回true），否则不连通（返回false）

核心算法代码（匹配例8.3）

            
int visited[MAXV]; // 全局访问标记数组
// 递归DFS遍历函数
void DFS(AdjGraph *G, int v)
{  ArcNode *p; int w;
   visited[v]=1; 		//置已访问标记
   printf("%d  ",v); 		//输出被访问顶点的编号
   p=G->adjlist[v].firstarc; 	//p指向顶点v的第一条边的边头结点
   while (p!=NULL)
   {  w=p->adjvex;
      if (visited[w]==0)
         DFS(G,w);   		//若w顶点未访问，递归访问它
      p=p->nextarc; 		//p指向顶点v的下一条边的边头结点
   }
}
// 连通性判断主函数
bool Connect(AdjGraph *G) {
    int i;
    bool flag = true;
    for (i=0; in; i++) // 访问标记置初值0
        visited[i] = 0;
    DFS(G, 0); // 从顶点0开始DFS遍历
    for (i=0; in; i++) { // 检查所有顶点访问状态
        if (visited[i] == 0) {
            flag = false; // 存在未访问顶点，图不连通
            break;
        }
    }
    return flag; // 返回连通性结果
}
