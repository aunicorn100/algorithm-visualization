// （2）邻接表转邻接矩阵算法
void ListToMat(AdjGraph *G, MatGraph &g)
//将邻接表G转换成邻接矩阵g
{  int i;
   ArcNode *p;
   for (i=0;in;i++)       //扫描所有的单链表
   {  p=G->adjlist[i].firstarc;   //p指向第i个单链表的首结点
      while (p!=NULL)             //扫描第i个单链表
      {  g.edges[i][p->adjvex]=p->weight;
         p=p->nextarc;
      }
   }
   g.n=G->n; g.e=G->e;
}
// 时间复杂度：O(n+e)（n为顶点数，e为边数）
// 仅访问所有头结点和边结点各一次
