算法代码（MatToList） 
// （1）邻接矩阵转邻接表算法
void MatToList(MatGraph g, AdjGraph *&G)
{   int i,j;
    ArcNode *p;
    G = (AdjGraph*)malloc(sizeof(AdjGraph));
    for (i=0; i<g.n; i++)       // 头结点指针域置空
        G->adjlist[i].firstarc = NULL;
    for (i=0; i<g.n; i++)       // 遍历所有顶点
        for (j=g.n-1; j>=0; j--) {
            if (g.edges[i][j] != 0 && g.edges[i][j] != INF) { // 存在有效边
                p = (ArcNode*)malloc(sizeof(ArcNode));  // 创建边结点
                p->adjvex = j;    // 存邻接点
                p->weight = g.edges[i][j];   // 存权值
                p->nextarc = G->adjlist[i].firstarc;   // 头插法
                G->adjlist[i].firstarc = p;
            }
        }
    G->n = g.n; G->e = g.e;
}
// 时间复杂度：O(n²)
