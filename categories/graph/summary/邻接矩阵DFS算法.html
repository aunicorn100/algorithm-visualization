#define MAXV 10
#define INF 32767
int visited[MAXV];

void MDFS(MGraph g, int v)
{  int w;
   printf("%d  ",v);
   visited[v]=1;

   for(w=0;w<g.n;w++)
      if(g.edges[v][w]!=0 && g.edges[v][w]!=INF && visited[w]==0)
         MDFS(g,w);
}
