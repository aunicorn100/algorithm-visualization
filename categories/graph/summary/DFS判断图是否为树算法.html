int visited[MaxSize];

void DFS2(ALGraph *G, int v, int &vn, int &en)
{  ArcNode *p;
   visited[v] = 1;
   vn++;
   p = G->adjlist[v].firstarc;

   while (p != NULL)
   {  en++;
      if (visited[p->adjvex] == 0)
         DFS2(G, p->adjvex, vn, en);
      p = p->nextarc;
   }
}

bool GIsTree(ALGraph *G)
{  int vn=0, en=0, i;
   for (i=0; i<MaxSize; i++)
      visited[i] = 0;
   DFS2(G, 0, vn, en);
   if (en == 2*(vn-1))
      return true;
   else
      return false;
}
