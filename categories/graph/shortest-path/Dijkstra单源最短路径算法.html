教材8.5.2 核心要点

  • 单源最短路径，边权≥0
  • S：已求出最短路径顶点集；U：未求出
  • 每步选U中dist最小的u加入S
  • 以u为中间点松弛更新dist
  • 顶点一旦入S，dist不再修改
  • 时间复杂度：O(n²)

Dijkstra算法代码（完全同PPT）


#define INF 32767		// 表示无穷大
#define MAXV 10		// 最大顶点数

void Dijkstra(MatGraph g, int v)
{  int dist[MAXV], path[MAXV];
   int s[MAXV];
   int mindis, i, j, u;

   for (i=0; i<g.n; i++)	// 初始化
   {  dist[i] = g.edges[v][i];
      s[i] = 0;
      if (g.edges[v][i] < INF)
         path[i] = v;
      else
         path[i] = -1;
   }
   s[v] = 1;			// 源点放入S

   for (i=0; i<g.n-1; i++)	// 循环 n-1 次
   {  mindis = INF;
      for (j=0; j<g.n; j++)	// 找 dist 最小的 u
         if (s[j]==0 && dist[j]<mindis)
         {  u=j; mindis=dist[j]; }
      s[u]=1;			// u 加入 S

      for (j=0; j<g.n; j++)	// 松弛操作
         if (s[j]==0)
            if (g.edges[u][j]<INF && dist[u]+g.edges[u][j]<dist[j])
            {  dist[j]=dist[u]+g.edges[u][j];
               path[j]=u;
            }
   }
   Dispath(dist, path, s, g.n, v);
}
