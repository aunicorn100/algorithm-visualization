// 直接插入排序算法
void InsertSort(RecType R[], int n)
{   int i, j;   RecType tmp;
    for (i=1; i<n; i++) {
        if (R[i].key < R[i-1].key) { // 反序时
            tmp = R[i];              // 保存待插入元素
            j = i - 1;
            do {                    // 找插入位置
                R[j+1] = R[j];      // 元素后移
                j--;
            } while (j>=0 && R[j].key > tmp.key);
            R[j+1] = tmp;           // 插入元素
        }
    }
}
// 时间复杂度：最好O(n)，最坏O(n²)
