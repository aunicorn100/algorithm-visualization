void MergeSortDC(RecType R[],int low,int high) 
//对R[low..high]进行二路归并排序
{  int mid;
   if (low<high)
   {  mid=(low+high)/2;
      MergeSortDC(R,low,mid);
      MergeSortDC(R,mid+1,high);
      Merge(R,low,mid,high);
   }
}

void MergeSort1(RecType R[],int n)	//自顶向下的二路归并算法
{
   MergeSortDC(R,0,n-1);
}
