void RadixSort(RecType1 *&p, int r, int d) {
    RecType1 *head[MAXR], *tail[MAXR], *t;
    int i, j, k;
    // 低位优先：从个位到百位（d-1 → 0）
    for (i = d - 1; i >= 0; i--) {
        // 初始化队列头/尾指针
        for (j = 0; j < r; j++) head[j] = tail[j] = NULL;
        // 分配阶段：按第i位分配到对应队列
        while (p != NULL) {
            k = p->data[i] - '0'; // 取当前位数字
            if (head[k] == NULL) { // 队列空，作为队头
                head[k] = p; tail[k] = p;
            } else { // 队列非空，追加到队尾
                tail[k]->next = p; tail[k] = p;
            }
            p = p->next; // 处理下一个节点
        }
        // 收集阶段：按队列顺序拼接回链表
        p = NULL; t = NULL;
        for (j = 0; j < r; j++) {
            if (head[j] != NULL) {
                if (p == NULL) { // 第一个非空队列
                    p = head[j]; t = tail[j];
                } else { // 追加到链表尾部
                    t->next = head[j]; t = tail[j];
                }
            }
        }
        if (t != NULL) t->next = NULL; // 链表收尾
    }
}
