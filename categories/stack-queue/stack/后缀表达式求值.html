速度：800 ms

算法核心思想

  • • 使用栈结构处理后缀表达式
  • • 遇到数字压入栈中
  • • 遇到运算符弹出两个操作数计算
  • • 计算结果重新压入栈中
  • • 最终栈中剩余元素即为结果

double compvalue(char *postexp)
{
   double d, a, b, c, e;
   SqStack1 *Opnd;
   InitStack1(Opnd);
   while (*postexp != '\0')
   {
      switch (*postexp)
      {
         case '+':
            Pop1(Opnd, a);
            Pop1(Opnd, b);
            c = b + a;
            Push1(Opnd, c);
            break;
         case '-':
            Pop1(Opnd, a);
            Pop1(Opnd, b);
            c = b - a;
            Push1(Opnd, c);
            break;
         case '*':
            Pop1(Opnd, a);
            Pop1(Opnd, b);
            c = b * a;
            Push1(Opnd, c);
            break;
         case '/':
            Pop1(Opnd, a);
            Pop1(Opnd, b);
            if (a == 0)
            {
               printf("\n\t除零错误！\n");
               exit(0);
            }
            c = b / a;
            Push1(Opnd, c);
            break;
         default:
            d = 0;
            while (*postexp >= '0' && *postexp <= '9')
            {
               d = 10 * d + (*postexp - '0');
               postexp++;
            }
            Push1(Opnd, d);
            break;
      }
      postexp++;
   }
   GetTop1(Opnd, e);
   DestroyStack1(Opnd);
   return e;
}
