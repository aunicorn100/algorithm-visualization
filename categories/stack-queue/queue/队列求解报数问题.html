void number(int n)
{
  int i; ElemType e;
  SqQueue *q;//环形队列指针q
  InitQueue(q);//初始化队列q
  for (i=1;i<=n;i++)//构建初始序列
    enQueue(q,i);
  printf("报数出列顺序:");
  while (!QueueEmpty(q))//队列不空循环
  {
    deQueue(q,e);//出队一个元素e
    printf("%d ",e);//输出元素编号
    if (!QueueEmpty(q))//队列不空
    {
      deQueue(q,e);//出队一个元素e
      enQueue(q,e);//将刚出列的元素进队
    }
  }
  printf("\n");
  DestroyQueue(q);//销毁队列
}
